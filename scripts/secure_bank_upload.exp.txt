#!/usr/bin/expect -f
# =============================================================================
# SCRIPT: secure_bank_upload.exp
# DESCRIPTION: Automates SFTP login using Key-Pair authentication
# =============================================================================

set timeout 30
set file_path [lindex $argv 0]
set category  [lindex $argv 1]

# Sanitized Connection Details
set host "sftp.partner-bank.com"
set user "REDACTED_CORP_USER"
set passphrase "REDACTED_KEY_PASSPHRASE"

# Initiate SFTP Connection
spawn sftp $user@$host

expect {
    "Enter passphrase for key*" {
        send "$passphrase\r"
        exp_continue
    }
    "sftp>" {
        # Navigate to bank-side directory based on payment category
        send "cd /incoming/payments/$category\r"
        expect "sftp>"
        
        # Upload the file
        send "put $file_path\r"
        expect "sftp>"
        
        send "exit\r"
    }
}

expect eof
